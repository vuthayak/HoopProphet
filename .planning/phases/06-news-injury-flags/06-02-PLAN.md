    """
    FastAPI router for news & injury flag endpoints — per D-09, D-10.

    GET /api/players/{player_id}/news — full alert details (news items, sources, timestamps, alert type, severity, raw headline/URL)
    No separate /api/news top-level endpoint per D-10.
    """

    from typing import Optional

    from fastapi import APIRouter, HTTPException, Request

    from server.services.news_service import NewsService
    from server.services.player_service import get_player_by_id
    from server.core.config import NEWS_STALE_WARNING_HOURS

    router = APIRouter(prefix="/api", tags=["news"])


    @router.get("/players/{player_id}/news")
    def get_player_news(player_id: int, refresh: bool = False):
        """Full news and alerts for a player.

        Per D-09: dedicated endpoint returning full alert details.
        Per D-07: includes "Updated X min/hours ago" timestamps.
        Per D-08: includes stale data warning if alerts are older than 24 hours.

        Query params:
        - refresh: if True, force re-fetch news from sources before returning
        """
        # Validate player exists
        try:
            player = get_player_by_id(player_id)
        except ValueError:
            raise HTTPException(status_code=404, detail="Player not found")

        news_service = NewsService()

        # Check cache freshness
        is_fresh, stale_sources = news_service.is_cache_fresh()

        # Force refresh if requested or if cache is stale
        if refresh or not is_fresh:
            news_service.process_all()

        # Get alerts and news items from DB
        from server.pipeline.db.connection import get_connection
        from server.pipeline.db.queries import get_player_alerts, get_news_items

        conn = get_connection()
        try:
            alerts = get_player_alerts(conn, player_id)
            news_items = get_news_items(conn, player_id, limit=50)
        finally:
            conn.close()

        # Format response per D-07 (timestamps) and D-08 (stale warning)
        now = ...  # current UTC time
        formatted_alerts = []
        for alert in alerts:
            last_updated = alert["last_updated_at"]
            # Compute "Updated X min/hours ago" per D-07
            age_delta = now - parse(last_updated)
            if age_delta.total_seconds() < 3600:
                updated_ago = f"{int(age_delta.total_seconds() / 60)} min ago"
            else:
                updated_ago = f"{int(age_delta.total_seconds() / 3600)} hours ago"

            formatted_alerts.append({
                "alert_type": alert["alert_type"],
                "subcategory": alert.get("subcategory"),
                "severity": alert["severity"],
                "source": alert["source"],
                "source_url": alert.get("source_url"),
                "headline": alert.get("headline"),
                "first_seen_at": alert["first_seen_at"],
                "last_updated_at": alert["last_updated_at"],
                "updated_ago": updated_ago,
            })

        formatted_news = []
        for item in news_items:
            formatted_news.append({
                "source": item["source"],
                "source_url": item.get("source_url"),
                "headline": item["headline"],
                "raw_content": item.get("raw_content"),
                "published_at": item["published_at"],
                "fetched_at": item["fetched_at"],
                "alert_keywords": item.get("alert_keywords"),
                "updated_ago": ...,  # same timestamp calculation
            })

        # Stale data warning per D-08
        stale_warning = None
        if any(alert for alert in formatted_alerts
               if ...age > NEWS_STALE_WARNING_HOURS):
            stale_warning = f"News data may be outdated. Last updated more than {NEWS_STALE_WARNING_HOURS} hours ago."

        return {
            "player_id": player_id,
            "alerts": formatted_alerts,
            "news_items": formatted_news,
            "stale_warning": stale_warning,
        }
    ```

    def get_player_alerts_summary(player_id: int) -> list[dict]:
        """Lightweight alert summary for embedding in player response per D-09.

        Returns only alert_type, severity, subcategory, and last_updated_at.
        Full details available via /api/players/{id}/news endpoint.
        """
        from server.pipeline.db.connection import get_connection
        from server.pipeline.db.queries import get_player_alerts

        conn = get_connection()
        try:
            alerts = get_player_alerts(conn, player_id)
        except Exception:
            # Graceful degradation — if alerts fail, return empty list
            return []
        finally:
            conn.close()

        return [
            {
                "alert_type": a["alert_type"],
                "severity": a.get("severity", "warning"),
                "subcategory": a.get("subcategory"),
                "last_updated_at": a["last_updated_at"],
            }
            for a in alerts
        ]
    ```

    Important: This function must NOT crash if the player_alerts table doesn't exist yet (graceful degradation). Wrap in try/except returning empty list.
