    compute_overall_calibration(predictions_df: pd.DataFrame, n_bins: int = CALIBRATION_BINS) -> Dict[str, Any]:
      - Takes the per-prediction DataFrame from backtest.py
      - Extracts y_true = predictions_df["hit"].values and y_pred = predictions_df["predicted_proba"].values
      - Delegates to metrics.compute_calibration_curve(y_true, y_pred, n_bins) for the curve
      - Also computes overall ECE, Brier score, log loss, accuracy, and total predictions
      - Returns dict with keys: fraction_positives, mean_predicted_value, bin_counts, ece, brier_score, log_loss, accuracy, n_predictions

    compute_per_stat_calibration(predictions_df: pd.DataFrame, n_bins: int = CALIBRATION_BINS) -> Dict[str, Dict[str, Any]]:
      - Per D-03: produces calibration curves grouped by stat type (pts, reb, ast, etc.)
      - Maps stat_type integer column to human-readable name using STAT_TYPE_NAMES reverse mapping
      - For each stat_type value, filters predictions_df and computes calibration curve
      - Skips stat types with fewer than 50 predictions (thin slices per D-03 deferred per-season calibration — flag them in output with "skipped: insufficient_samples")
      - Returns dict keyed by stat name string (e.g. "pts", "reb")

    compute_season_breakdown(predictions_df: pd.DataFrame) -> Dict[str, Dict[str, float]]:
      - Per TEST-03: season-by-season accuracy breakdown
      - Groups predictions_df by "season" column
      - Per D-04: full report with summaries
      - For each season, computes: accuracy, log_loss, brier_score, n_predictions, mean_predicted_proba, hit_rate (observed)
      - Also computes "overall" aggregate across all seasons
      - Handles thin seasons gracefully (empty groups or <10 predictions are still reported but flagged)
      - Returns dict keyed by season string (e.g. "2022-23")

    compute_roi_metrics(predictions_df: pd.DataFrame, threshold: float = BREAKEVEN_THRESHOLD) -> Dict[str, Any]:
      - Per TEST-04 and D-02: vig-adjusted ROI using -110 odds
      - For each prediction where predicted_proba > threshold (default 52.4%):
        - If hit == 1: profit = 0.909 units (win $100 on a $110 bet = 100/110 = 0.909)
        - If hit == 0: profit = -1.0 units (lose the $110 bet, normalized to -1.0 unit per unit wagered)
        - Net units = sum of all profits; total_wagered = number of bets * 1.0 (each bet is 1.0 unit)
        - ROI = net_units / total_wagered
      - Also compute ROI by confidence bucket (e.g. 52-55%, 55-60%, 60-70%, 70-80%, 80%+)
      - Returns dict with: overall_roi, roi_by_bucket, total_bets, win_count, loss_count, net_units, total_wagered

    compute_confidence_intervals(predictions_df: pd.DataFrame, metric_funcs: Dict[str, Callable] = None, n_bootstrap: int = BOOTSTRAP_SAMPLES, ci_level: float = CONFIDENCE_LEVEL) -> Dict[str, Dict[str, float]]:
      - Per D-05: bootstrap confidence intervals
      - Default metric_funcs provides: accuracy, brier_score, log_loss, roi (at BREAKEVEN_THRESHOLD)
      - For each metric: resample predictions_df rows with replacement n_bootstrap times, compute metric on each resample
      - Return low (2.5th percentile), mid (50th), and high (97.5th percentile) bounds for each metric
      - Uses numpy.percentile for bootstrap CI bounds
      - Returns dict like: {"accuracy": {"low": 0.54, "mid": 0.58, "high": 0.62}, ...}

    IMPORTANT: All functions take the per-prediction DataFrame as input. The DataFrame schema matches Plan 01's BacktestResult.predictions_df. Since both plans are in Wave 1, this contract is defined here and Plan 01's implementation must match it.
